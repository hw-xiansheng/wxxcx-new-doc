3、methods定义获取数据方法


 
 
 
 
 
 
 
 
 
 
 
 


methods: {
    // 2. 获取首页数据
    async getIndexData(){
        // 2.1 请求数据
        let  res = await uni.$http.get('/index/appInfo')
        // 2.2 赋值变量
        this.banner = res.banner
        this.categoryList = res.categoryList
        this.channel = res.channel
        this.notice = res.notice.map((item,index)=>{
            return (index+1)+'.'+item.content+' '
        }).join('')
    }
}

注意： 在utils/request.js的响应拦截器中进行对数据解构处理一下，集中能处理错误，和返回解构数据



 
 
 
 
 
 
 
 
 
 
 
 




// 响应拦截
uni.$u.http.interceptors.response.use((response) => {
	const { errno,errmsg,data } = response.data 
	if(errno === 0){
		// 业务正确
		return data
	}else{
		// 业务错误
		uni.showToast({
			icon:'none',
			title:errmsg
		})
		return Promise.reject(errmsg)
	}
}, (response) => { 
	return Promise.reject(response)
})
